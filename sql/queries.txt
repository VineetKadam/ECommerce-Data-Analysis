#total revenue

SELECT SUM("Revenue") AS total_revenue
FROM sales_data;


# top 5 products by revenue

SELECT "Product Name", SUM("Revenue") AS total_revenue
FROM sales_data
GROUP BY "Product Name"
ORDER BY total_revenue DESC
LIMIT 5;


# sales by region

SELECT "Region", SUM("Revenue") AS total_revenue
FROM sales_data
GROUP BY "Region"
ORDER BY total_revenue DESC;


# monthly sales trend

SELECT "Month", SUM("Revenue") AS monthly_sales
FROM sales_data
GROUP BY "Month"
ORDER BY "Month";


# top customers

SELECT "Customer Name", SUM("Revenue") AS total_spent
FROM sales_data
GROUP BY "Customer Name"
ORDER BY total_spent DESC
LIMIT 5;
